Aprendendo sobre banco de dados

Principais Comandos Banco

CREATE DATABASE nome_do_banco;

DROP DATABASE nome_do_banco;

USE nome_do_banco;

SHOW DATABASES;

CREATE TABLE nome_da_tabela (coluna1 tipo, coluna2 tipo, coluna3 tipo);

DROP TABLE nome_da_tabela;

SHOW TABLES;

INSERT INTO nome_da_tabela (coluna1, coluna2, coluna3) VALUES (valor1, valor2, valor3);

SELECT * FROM nome_da_tabela;

SELECT coluna1, coluna2 FROM nome_da_tabela;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1 AND coluna2 = valor2;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1 OR coluna2 = valor2;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1 ORDER BY coluna2;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1 ORDER BY coluna2 DESC;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1 LIMIT 10;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1 LIMIT 10 OFFSET 5;

SELECT * FROM nome_da_tabela WHERE coluna1 = valor1 LIMIT 10 OFFSET 5;

SELECT COUNT(*) FROM nome_da_tabela;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1 AND coluna2 = valor2;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1 OR coluna2 = valor2;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1 ORDER BY coluna2;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1 ORDER BY coluna2 DESC;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1 LIMIT 10;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1 LIMIT 10 OFFSET 5;

SELECT COUNT(*) FROM nome_da_tabela WHERE coluna1 = valor1 LIMIT 10 OFFSET 5;

SELECT SUM(coluna1) FROM nome_da_tabela;

SELECT SUM(coluna1) FROM nome_da_tabela WHERE coluna2 = valor2;

SELECT AVG(coluna1) FROM nome_da_tabela;

SELECT AVG(coluna1) FROM nome_da_tabela WHERE coluna2 = valor2;

SELECT MAX(coluna1) FROM nome_da_tabela;

SELECT MAX(coluna1) FROM nome_da_tabela WHERE coluna2 = valor2;

SELECT MIN(coluna1) FROM nome_da_tabela;

SELECT MIN(coluna1) FROM nome_da_tabela WHERE coluna2 = valor2;

SELECT coluna1, COUNT(*) FROM nome_da_tabela GROUP BY coluna1;

SELECT coluna1, COUNT(*) FROM nome_da_tabela WHERE coluna2 = valor2 GROUP BY coluna1;

SELECT coluna1, COUNT(*) FROM nome_da_tabela WHERE coluna2 = valor2 GROUP BY coluna1 HAVING COUNT(*) > 1;

SELECT coluna1, COUNT(*) FROM nome_da_tabela WHERE coluna2 = valor2 GROUP BY coluna1 HAVING COUNT(*) > 1 ORDER BY COUNT(*) DESC;

SELECT coluna1, COUNT(*) FROM nome_da_tabela WHERE coluna2 = valor2 GROUP BY coluna1 HAVING COUNT(*) > 1 ORDER BY COUNT(*) DESC LIMIT 10;
